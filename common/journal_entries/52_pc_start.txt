je_pc_start = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_internal_affairs

	possible = {
		player = this
	}
	is_shown_when_inactive = {
		always = yes
		this = c:GBR
	}

	immediate = {
		trigger_event = pc_start.1
	}

	timeout = 14

	on_timeout = {
		play_as = c:PCM
		c:PCM = {
			activate_law = law_type:law_pc_corp_economy
			create_treaty = {
					binding_period = { years = 200 }
					articles_to_create = { 
						{
							article = foreign_investment_rights
							source_country = ROOT
							target_country = c:PCM
						}
					}
					first_country = c:PCM
					second_country = root
					is_draft = no
				}
				add_journal_entry = {
					type = je_pc_update_modifers
				}
			set_variable = {
				name = pc_num_free_corp_building_type_var
				value = 1
			}
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}
