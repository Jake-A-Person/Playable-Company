je_pc_init_fake_company = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_internal_affairs

	possible = {
		this = c:PCM
	}
	
	complete = {
		any_company = {
			is_company_type = company_type:company_pc_fake
		}
	}

	on_complete = {
		add_journal_entry = {
			type = je_pc_update_fake_company
		}
	}

	weight = 100
	should_be_pinned_by_default = yes
}

je_pc_update_fake_company = {
	icon = "gfx/interface/icons/event_icons/event_military.dds"

	group = je_group_internal_affairs

	immediate = {
		#finds the building that was created in pc_start that is in the country that is being invested in
		random_state = {
			limit = {
				not = {
					owner = root
				}
				has_building = building_pc_corp_hq_add_money
			}
			#change building ownership by removing the building and making a new one with correct ownership
			remove_building = building_pc_corp_hq_add_money
			create_building = {
				reserves = 100
				building = building_pc_corp_hq_add_money
				add_ownership = {
					company = {
						type = company_pc_fake
						country = ROOT
						levels = 1
					}
				}
			}
			random_scope_building = {
				limit = {
					is_building_type = building_pc_corp_hq_add_money
				}
				#sets the building profit to be the player company profit
				add_modifier = pc_hq_add_money_offset
				save_scope_as = scope_pc_company_hq_add_money
			}
		}
		#make sure the executive has no modifers that will effect building profit
		random_scope_character = {
			limit = {
				has_role = executive
			}
			trigger_event = pc_clean_executive.1
		}
	}

	on_weekly_pulse = {
		effect = {
			scope:scope_pc_company_hq_add_money = {
				#update the profit of the fake money building to be in line with
				#current profit of player company
				add_modifier = {
					name = pc_hq_add_money_income_mult
					multiplier = ROOT.weekly_net_fixed_income
					#with 1 week will still be active on day 7 so the modifer would be on twice for 1 day
					days = 6
				}
			}
		}
	}

	on_yearly_pulse = {
		effect = {
			#make sure the executive has no modifers that will effect building profit
			random_scope_character = {
				limit = {
					has_role = executive
				}
				trigger_event = pc_clean_executive.1
			}
		}
	}
	weight = 100
	should_be_pinned_by_default = yes
}
