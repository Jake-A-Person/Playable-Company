pc_corp_building_spec_coal_scripted_button = {
	name = "pc_corp_building_spec_coal_scripted_button"
	desc = "pc_corp_building_spec_coal_scripted_button_desc"

	visible = { or = { 
		has_variable = pc_corp_building_spec_coal_var
		and = {
			has_variable = pc_num_free_corp_building_type_var
			var:pc_num_free_corp_building_type_var > 0
		}
	}
	}
	
	possible = {
		not = {
			custom_tooltip = {
				text = pc_corp_building_is_spc_tt
				has_variable = pc_corp_building_spec_coal_var
			}
		}
	}

	effect = {
		set_variable = pc_corp_building_spec_coal_var
		change_variable = {
			name = pc_num_free_corp_building_type_var
			add = -1
		}
	}
}
