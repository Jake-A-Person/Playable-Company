namespace = pc_add_corp_building_type

#on research
pc_add_corp_building_type.1 = {
	type = country_event

	hidden = yes 
	
	trigger = {
		this = c:PCM
		has_technology_researched = corporatism
	}

	cooldown = { years = 200 }

	immediate = {
		change_variable = {
			name = pc_num_free_corp_building_type_var
			add = 1
		}
	}
}
